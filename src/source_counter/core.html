(ns source-counter.core (:use clojure.java.io) (:use clojure.pprint))

(defmacro dbg [x] `(let [x# ~x] (println "dbg:" '~x "=" x#) x#))

(defn read-file "Reads a file line by line and returns it as a vector of lines." [path] (with-open [rd (reader path)] (reduce conj [] (line-seq rd))))

(defn check-file-exts "For a vector of file extensions, checks if name matches either of them." [name extensions] (reduce #(or %1 %2) (map

#(.endsWith name %) extensions)))

(defn get-file-ext "Returns the file extension of filename." [filename] (str "." (last (clojure.string/split filename #"\."))))

(defn find-files "Returns a tree-seq of all files that have one of the file extensions that are passed in the extensions vector." [path extensions] (filter

#(check-file-exts (.getName %) extensions) (file-seq (file path))))

(defn count-loc "Counts the lines of code, ignores lines that return false for check-line (a supplied predicate function for a string)" [file-lines check-line] (reduce + (map #(if (check-line %) 1 0) file-lines)))

; A few predicate functions to check a source line of a given language (defn check-clojure-line [l] (let [line (.trim l)] (if (or (.startsWith line ";") (.isEmpty line)) false true)))

(defn check-c-ish-line [l] (let [line (.trim l)] (if (or (.startsWith line "//") (.startsWith line "/*") (.startsWith line "*") (.isEmpty line)) false true)))

(defn check-py-line [l] (let [line (.trim l)] (if (or (.startsWith line "#") (.isEmpty line)) false true)))

(def lang-map [[[".clj"] check-clojure-line] [[".scala" ".cs" ".c" ".cpp" ".java" ".h" ".hpp"] check-c-ish-line] [[".rb" ".py"] check-py-line]])

(def supported-languages (reduce (fn [vec [exts _]] (into vec exts)) [] lang-map))

(def lang-names {".clj" "Clojure" ".scala" "Scala" ".c" "C" ".cpp" "C++" ".hpp" "C++ Header" ".h" "C/C++ Header" ".rb" "Ruby" ".py" "Python" ".cs" "C#" ".java" "Java"})

(defn seq-contains? "Returns whether coll contains a target value." [coll target] (some #(= target %) coll))

(defn first-non-nil "Returns the first non-nil element in coll." [coll] (if (empty? coll) [] (loop [c coll] (if (first c) (first c) (recur (rest c))))))

(defn get-from-lang-map [e] (first-non-nil (for [[exts method] *lang-map*] (when (seq-contains? exts e) method))))

(defn count-loc-in-dir "Returns a map from file extensions to LoC counts. Counts all of the files in the directory and recursively counts all other directories within that one too." [path] (let [files (find-files path supported-languages) data (map (fn [file] [(get-file-ext (.getName file)) (read-file file)]) files)] (reduce (fn [map [ext code]] (let [counter (get map ext 0) lines (count-loc code (get-from-lang-map ext))] (assoc map ext (+ lines counter)))) {} data)))

(defn make-loc-str [loc-map] (clojure.string/join "\r\n" (for [[name count] loc-map] (str (get lang-names name "Unknown language") ": " count " LoC"))))

(println (make-loc-str (count-loc-in-dir "D:/Dokumente/IdeaProjects")))
